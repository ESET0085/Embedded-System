#include #include #define LED_PIN PD6 #include int main(void) { // Set PD5 as output (LED) DDRD |= (1 << PD5); // Set PD2 as input (Button) DDRD &= ~(1 << PD2); // Enable pull-up resistor on PD2 PORTD |= (1 << PD2); while (1) { if (!(PIND & (1 << PD2))) // If button is pressed (active LOW) { PORTD |= (1 << PD5); // Turn ON LED } else { PORTD &= ~(1 << PD5); // Turn OFF LED } } }
